@tab pnpm

```sh
#在Yunzai-Bot根目录执行
git clone --depth=1 https://github.com/yeyang52/yenai-plugin.git ./plugins/yenai-plugin

git clone --depth=1 https://github.com/yeyang52/yenai-plugin.git ./plugins/yenai-plugin

#安装依赖
npm install systeminformation cheerio -w
```
